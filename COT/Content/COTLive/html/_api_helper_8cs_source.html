65  {
66  var json = wc.DownloadString(url);
67 
68  JObject parsed = JObject.Parse(json);
69 
70  Logging(Status.MakeCall, Convert.ToInt32(parsed["hits"]["total"]));
71  foreach (var obj in parsed["hits"]["hits"].Children())
72  {
73  // Json properties to variables
74  int elasticSearchId = Convert.ToInt32(obj["_id"].ToString());
75  DateTime dateCreated = DateTime.Parse(obj["fields"]["date_created"][0].ToString());
76  DateTime dateLastAction = DateTime.Parse(obj["fields"]["date_last_action"][0].ToString());

79  string requestStatus = obj["fields"]["request_status"][0].ToString();
80  string requestType = obj["fields"]["request_type"][0].ToString();
81  string origin = obj["fields"]["origin"][0].ToString();
82  double latitude = Convert.ToDouble(obj["fields"]["latitude"][0].ToString());
83  double longitude = Convert.ToDouble(obj["fields"]["longitude"][0].ToString());
84 
85  // Create new Query
86  Query newCall = new Query
87  {
88  ElasticSearchId = elasticSearchId,
89  DateCreated = dateCreated,
90  DateLastAction = dateLastAction,
91  District = district,
92  Department = department,
93  RequestStatus = requestStatus,
94  RequestType = requestType,
95  Origin = origin,
96  Latitude = latitude,
97  Longitude = longitude
98  };
99 
100  // If the new record's date of last action is older than a set date and the status is closed, we do not want to re-add it.
101  // If the record is open or inprogress
102  if ((newCall.DateLastAction < _olderThan && newCall.RequestStatus != "Closed") ||
103  newCall.DateLastAction > _olderThan)
104  {
105  calls.Add(newCall);
106  }
107  }
108 
109  }

153  public void CheckDuplicate()
154  {
155  using (var ctx = new ApplicationContext())
156  {
157  var removeFromNew = new List<Query>();
158  // Int list because only getting the ElasticSearch ID
159  var removeFromOld = new List<int>();
160 
161  // Get all records in dB
162  var query = ctx.Queries.ToList();
163 
164  // Compare the new data to what is currently in the database
165  foreach (var record in calls)
166  {
167  // If there is a record with the same ElasticSearchId
168  if (query.Any(x => x.ElasticSearchId == record.ElasticSearchId))
169  {
170  // Select the first or default record containing the elasticSearchId
171  var item = query.FirstOrDefault(x => x.ElasticSearchId == record.ElasticSearchId);
172  // If the dates match then it is a duplicate
173  if (item != null && record.DateLastAction.Equals(item.DateLastAction))
174  {
175  removeFromNew.Add(record);
176  }
177  // Otherwise it is not a duplicate and we should take the latest one.
178  else if (item != null && record.DateLastAction > item.DateLastAction)
179  {
180  // Remove
181  removeFromOld.Add(item.ElasticSearchId);
182  }
183  }
184  }
185 
186  // Logging num
187  int numRemoved = removeFromNew.Count + removeFromOld.Count;
188 
189  // Removes records from the new dataset if it is the same as a record in the database.
190  if (removeFromNew.Count > 0)
191  {
192  calls.RemoveAll(x => removeFromNew.Exists(y => y.ElasticSearchId == x.ElasticSearchId));
193  }
194 
195  // Removes record from the database when a new one of the same ElasticSearchID is incoming in the new data set.
196  if (removeFromOld.Count > 0)
197  {
198  foreach (var id in removeFromOld)
199  {
200  ctx.Queries.Remove(ctx.Queries.First(x => x.ElasticSearchId == id));
201  }
202  ctx.SaveChanges();
203  }
204  Logging(Status.CheckDuplicate, numRemoved);
205  }
206  }

241  private static void Logging(Status status, int records = 0, bool connected = false, string error = null )
242  {

245  var dateline = $"[{DateTime.Now}]\n";

252  switch (status)

273  break;

279  break;

282  newLine = $"[Truncate] {records} removed from DB\n";

284  newLine = $"[{DateTime.Now}] [Truncate] {records} older than {_olderThan} removed from the database.\n";
285  break;
286  case Status.Error:
287 
288  // Send an error email
289  if (_settings.ErrorNotificationEnabled)
290  {

301  Logging(Status.Error, error: $"[Email] {ex.Message}");
302 
303  }
304  }
305 
306  if (_verbosity == Verbosity.Minimal)
307  newLine = $"[Error] {error}";
308  if (_verbosity == Verbosity.Verbose)
309  newLine = $"[{DateTime.Now}] [Error] {error}\n";
310  break;
311  }
312 
313  // Adds log msg to logLines list to be added as a batch later.
314  logLines.Add(newLine);
